def insertDict(dictToInsert, globalDict):
    for key, value in dictToInsert.items():
        if key in globalDict:
            globalDict[key] += value
        else:
            globalDict[key] = value
    return globalDict

d3 = insertDict(d1, d3)
d3 = insertDict(d2, d3)

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    # prints the point out
    def show(self):
        print("(" + str(self.x) + ", " + str(self.y) + ")")
    
    def dist(self, otherPoint):
        xDist = abs(self.x - otherPoint.x)
        yDist = abs(self.y - otherPoint.y)
        return math.sqrt(xDist ** 2 + yDist ** 2)


# demonstration of functions
p1 = Point(0, 0)
p2 = Point(10, 10)

p1.show()
p2.show()
print(p1.dist(p2))
print(p2.dist(p1))

(0, 0)
(10, 10)
